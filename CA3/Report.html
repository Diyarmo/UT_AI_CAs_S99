      def nltk2wn_tag(nltk_tag):
          if nltk_tag.startswith('J'):
              return wordnet.ADJ
          elif nltk_tag.startswith('V'):
              return wordnet.VERB
          elif nltk_tag.startswith('N'):
              return wordnet.NOUN
          elif nltk_tag.startswith('R'):
              return wordnet.ADV
          else:                    
              return None

      def stem_clean(corpus):
          corpus = " ".join(corpus).lower()
          corpus = " ".join(map(lambda x: ps.stem(x), nltk.word_tokenize(corpus)))

          corpus = clean_stop_words(corpus)

      class Classifier:        
          def __init__(self):
              self.log_priors = None
              self.counters = []
              self.lemm = True
              self.num_words = []
              
          def fit(self, X, Y, lemm = None, num_classes = 3):
              if lemm != None:
                  self.lemm = lemm
              classes = ['TRAVEL', 'BUSINESS', 'STYLE & BEAUTY'][:num_classes]
              counts = []
              for cat in classes:
                  corpus = (X[Y == cat]).str.lower()
                  counts.append(len(corpus))
                  if self.lemm:
                      corpus = lemm_clean(corpus)
                  else:
                      corpus = stem_clean(corpus)
                  counter = Counter(corpus.split())
                  self.counters.append(counter)
                  self.num_words.append(sum(counter.values()))
                  
              self.log_priors = np.log(np.array(counts) / np.sum(counts))
      #         self.log_priors = np.zeros(num_classes)
          def predict(self, X):
              predicts = []
              for desc in X:
                  if self.lemm:
                      corpus = lemm_clean([desc.lower()])
                  else:
                      corpus = stem_clean([desc.lower()])
                      
                  counter = Counter(corpus.split())
                  scores = self.log_priors.copy()
      ​
                  for i in range(len(scores)):
                      score = list(map(lambda x:counter[x] * np.log1p(self.counters[i][x]), counter))
                      scores[i] += np.sum(score) - np.log(self.num_words[i]) * len(score)
                  predicts.append(np.argmax(scores))
      ​
              return predicts

      ## OverSampling

      OverSampling¶

      In [ ]:

      . . .
      x
       
      Unbalanced number of samples from each class leads to greater prior in crowded class, thus samples are more likely to classified as the crowded class.
      ​
      This may lead to a difference in precision and recall.
      ​
      There are a few approaches to address this problem:
      ​
      1- Remove some samples from crowded class[es] so all classes have almost equal number of samples.(in case of small datasets, we may lose some important samples)
      ​
      2- Augmentation and generating new samples from existing samples of less crowded classes.
      ​
      3- Since both above approaches are trying to reach equal priors by having equal number of samples in each class, we don't expect probabilities to change dramatically. We can easily set all priors to be equal and multply rows of confusion matrix to have equal sum. It's the same as both above approaches.

      ## 2- tf-idf

      2- tf-idf¶

      xxxxxxxxxx
       
      It stands for term frequency inverse document frequency.
      ​
      It's a way to show importance of a word for a document within a corpus.
      by using it we don't have to take care of stop words, because they exist in all documents, they are not imporant to any of documents. On the other hand rare words play an important role.
      ​
      We can use values of tf-idf table as wights for probabilities in our model. By doing so, we are decreasing the effect of stop-words and other common words in the corpus, and increasing effect of rare and important words.
